7.7 Simplified ggplot2

ggplot(data = faithful, mapping = aes(x = eruptions)) + 
  geom_freqpoly(binwidth = 0.25)

ggplot(faithful, aes(eruptions)) + 
  geom_freqpoly(binwidth = 0.25)

# Or Jon's crazy way
ggplot(faithful) +
  aes(eruptions) +
  geom_freqpoly(binwidth = 0.25)
